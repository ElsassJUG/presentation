\begin{frame}
    \frametitle{L'association}
    \begin{center}
        \begin{itemize}
        \itemcol Une association loi 1900 à but non lucratif et non commercial.
        \itemcol L'objectif de l'Elsass JUG est d'animer la communauté des
            développeurs en Alsace.
        \itemcol L'association est un lieu de rencontres et d'échanges.
        \itemcol Son objet premier est d'organiser régulièrement des
            évènements thématiques.
        \itemcol L'association se réserve la possibilité de prendre part à 
            d'autres initiatives visant au développement de la communauté 
        informatique en Alsace, seule ou en partenariat avec d'autres organisations.
        \itemcol Pas de cotisation !
        \end{itemize}
    \end{center}
\end{frame}
